# 发布订阅模式
```ts
Interface IEmitter {
    once: () => void
    on: () => void
    off: () => void
    emit: () => void
}

class Emitter implements IEmitter{
    events: Map<string,Function[]>
    constructor() {
        this.events = new Map()
    }
    once(event: string, fn: Function) {
        const cb = () => {
            fn()
            this.off(event, fn)
        }
        this.on(event, fn)
    }
    on(event: string, fn: Function): void {
        //是否是第一次存
        if(this.event.has(event)) {
            const callbackList = this.event.get(event)
            callbackList && callbackList.push(fn)
        } else {
            this.event.set(event, [fn])
        }
    }
    off(event: sting, fn: Function): void {
        if(this.event.has(event)) {
            const callbackList = this.event.get(event)
            callbackList && callbackList.splice(callbackList.indexOf(fn),1)
        }
    }
    emit(event: string, ...args: any[]): void {
        if(this.event.has(event)) {
            const callbackList = this.event.get(event)
            callbackList.forEach(callback => {
                callback(args)
            })
        }
    }
}

const emitter = new Emitter()

const fn = (a, b, c) => {
    console.log(a, b, c)
}

emitter.on('xiaoteng', fn)

emitter.off('xiaoteng', fn)
```
